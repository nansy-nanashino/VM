// 初期化（設定ファイル作成）
$ vagrant init
//IPの設定
config.vm.network "private_network", ip: “***.***.**.**”
//存在していないVMをVBから削除する
$ vagrant global-status --prune
//yumリストを"httpd"で検索
$ yum list | grep httpd
//インストールコマンド
$ sudo yum install
//インストール済みパッケージの確認
$ yum list installed | grep httpd
//VMを抜ける
control+D
//vagrantを止める
$vagrant halt
